import os import requests from bs4 import BeautifulSoup import time import re import urllib.parse # 1. 갤러리 이미지를 안전하게 저장할 폴더 생성 os.makedirs('assets/gallery', exist_ok=True) gallery_data = [] img_counter = 1 print("🚀 알려주신 고유 번호(35~2번)로 원본 사진 다이렉트 추출을 시작합니다...\n") # 💡 핵심: 목록 페이지를 거치지 않고, 35번부터 2번까지 직접 접속합니다! for wr_id in range(35, 1, -1): url = f"http://laibi.yonsei.ac.kr/bbs/board.php?bo_table=sub4_1&wr_id={wr_id}" try: response = requests.get(url) response.encoding = 'utf-8' soup = BeautifulSoup(response.text, 'html.parser') # 삭제된 글이거나 에러 페이지면 건너뛰기 if "존재하지 않" in response.text or "삭제된" in response.text: continue # 1. 제목 추출 title_tag = soup.find(class_=re.compile('bo_v_tit|title', re.I)) if title_tag: title = title_tag.get_text(strip=True).replace('\n', ' ') else: # 적절한 태그가 없으면 브라우저 탭 이름()에서 가져오기 title = soup.title.get_text(strip=True).split('>')[0].split('|')[0].strip() if soup.title else f"Gallery {img_counter}" # 2. 고화질 원본 사진 링크 추출 (그누보드 공식 해제) high_res_url = None # 패턴 1: 원본 보기 링크 (view_image.php?fn=진짜파일명.jpg) view_img_link = soup.find('a', href=re.compile(r'view_image\.php\?.*fn=([^&"\']+)')) if view_img_link: fn_match = re.search(r'fn=([^&"\']+)', view_img_link['href']) if fn_match: fn = fn_match.group(1) # 썸네일(thumb-)이 아닌 진짜 원본 파일 주소 high_res_url = f"http://laibi.yonsei.ac.kr/data/file/sub4_1/{fn}" # 패턴 2: 원본 보기 링크가 없는 경우, 본문 이미지 중 가장 큰 것 if not high_res_url: content_area = soup.find(id='bo_v_con') or soup.body if content_area: first_img = content_area.find('img', src=re.compile(r'data/file/sub4_1/')) if first_img: high_res_url = first_img['src'] if not high_res_url.startswith('http'): high_res_url = urllib.parse.urljoin("http://laibi.yonsei.ac.kr", high_res_url) # 만약 썸네일 경로라면 원본 경로로 억지로 변환 if 'thumb-' in high_res_url: # 예: thumb-xxx_600x400.jpg -> xxx.jpg 로 변환하는 로직 high_res_url = re.sub(r'/thumb-([^_]+)_.+', r'/\1', high_res_url) # 3. 사진 다운로드 및 저장 if high_res_url: ext = high_res_url.split('.')[-1][:3].lower() if ext not in ['jpg', 'png', 'gif', 'jpe']: ext = 'jpg' local_filename = f"assets/gallery/photo_{img_counter:03d}.{ext}" print(f"[{wr_id}번 게시글] '{title}' 원본 다운로드 중...") img_data = requests.get(high_res_url).content with open(local_filename, 'wb') as f: f.write(img_data) gallery_data.append({ "title": title, "image": local_filename }) img_counter += 1 except Exception as e: print(f" -> 에러 발생 (게시글 {wr_id}번): {e}") time.sleep(0.5) # 서버 보호 # 자바스크립트 파일(gallery_data.js) 생성 js_content = "const photos = [\n" for item in gallery_data: title_esc = item['title'].replace("'", "\\'") js_content += f" {{ title: '{title_esc}', image: '{item['image']}' }},\n" js_content += "];\n" with open("gallery_data.js", "w", encoding="utf-8") as f: f.write(js_content) print(f"\n✅ 완벽합니다! 총 {len(gallery_data)}개의 원본 사진을 'assets/gallery/'에 저장 완료했습니다.")
